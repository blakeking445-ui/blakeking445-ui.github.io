// Scene const scene = new THREE.Scene(); scene.background = new THREE.Color(0xeeeeee); // Camera (FIXED for better zoom) const camera = new THREE.PerspectiveCamera( 60, window.innerWidth / window.innerHeight, 0.001, // super close zoom 1000 ); camera.position.set(0.1, 0.1, 0.2); // Renderer const renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); // Controls (BETTER ZOOM SETTINGS) const controls = new OrbitControls(camera, renderer.domElement); controls.enableDamping = true; controls.minDistance = 0.001; // allows zoom VERY close controls.maxDistance = 100; controls.target.set(0, 0, 0); // Lighting const light1 = new THREE.DirectionalLight(0xffffff, 1); light1.position.set(5, 5, 5); scene.add(light1); const light2 = new THREE.AmbientLight(0xffffff, 0.6); scene.add(light2); // Loader const loader = new GLTFLoader(); let parametricModel, scanModel; // Load parametric loader.load("baseparametric.glb", (gltf) => { parametricModel = gltf.scene; parametricModel.traverse(child => { if (child.isMesh) { child.material.transparent = true; child.material.opacity = 0.7; child.material.color.set(0x0077ff); } }); scene.add(parametricModel); }); // Load scan loader.load("basescan.glb", (gltf) => { scanModel = gltf.scene; scanModel.traverse(child => { if (child.isMesh) { child.material.transparent = true; child.material.opacity = 0.7; child.material.color.set(0xff5500); } }); scene.add(scanModel); }); // Toggle functions window.toggleParametric = () => { if (parametricModel) parametricModel.visible = !parametricModel.visible; }; window.toggleScan = () => { if (scanModel) scanModel.visible = !scanModel.visible; }; // Resize window.addEventListener("resize", () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); }); // Animate function animate() { requestAnimationFrame(animate); controls.update(); renderer.render(scene, camera); } animate();
